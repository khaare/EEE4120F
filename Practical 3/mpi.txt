/*
 * Wariara Freights Inc.
 * Distributed Memory Branch-and-Bound Solver (MPI)
 *
 * Compile :
 *   mpicc -O2 -o wariara_mpi wariara_freights_mpi_full.c
 *
 * Run :
 *   mpirun -np 4 ./wariara_mpi -i input.txt
 */

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <limits.h>
#include <stdbool.h>
#include <time.h>
#include <mpi.h>

/* ───────────────────────── Timing ───────────────────────── */
double gettime(void)
{
    struct timespec ts;
    clock_gettime(CLOCK_MONOTONIC, &ts);
    return (double)ts.tv_sec + (double)ts.tv_nsec * 1e-9;
}

/* ───────────────────────── Problem Data ───────────────────────── */
#define MAXN 20

int N;
int E[MAXN][MAXN];

int best_cost = INT_MAX;
int best_path[MAXN];

bool visited[MAXN];

long long routes_completed = 0;
long long routes_pruned    = 0;

/* ───────────────────────── Factorial ───────────────────────── */
long long factorial(int n)
{
    long long r = 1;
    for (int i = 2; i <= n; i++)
        r *= i;
    return r;
}

/* ───────────────────────── Branch & Bound ───────────────────────── */
void branch_and_bound(int level,
                      int current_city,
                      int current_cost,
                      int path[])
{
    if (current_cost >= best_cost) {
        routes_pruned++;
        return;
    }

    if (level == N) {
        routes_completed++;
        if (current_cost < best_cost) {
            best_cost = current_cost;
            for (int i = 0; i < N; i++)
                best_path[i] = path[i];
        }
        return;
    }

    for (int next = 0; next < N; next++) {
        if (!visited[next]) {
            visited[next] = true;
            path[level] = next;

            branch_and_bound(level + 1,
                             next,
                             current_cost + E[current_city][next],
                             path);

            visited[next] = false; /* backtrack */
        }
    }
}

/* ───────────────────────── Input ───────────────────────── */
void read_input(const char *filename)
{
    FILE *f = fopen(filename, "r");
    if (!f) {
        fprintf(stderr,"Error opening %s\n", filename);
        MPI_Abort(MPI_COMM_WORLD,1);
    }

    fscanf(f,"%d",&N);

    for(int i=0;i<N;i++)
        for(int j=0;j<N;j++)
            E[i][j]=0;

    for(int j=1;j<N;j++)
        for(int i=0;i<j;i++){
            fscanf(f,"%d",&E[i][j]);
            E[j][i]=E[i][j];
        }

    fclose(f);
}

/* ───────────────────────── MAIN ───────────────────────── */
int main(int argc, char *argv[])
{
    double t_init_start = gettime();

    MPI_Init(&argc,&argv);

    int rank, size;
    MPI_Comm_rank(MPI_COMM_WORLD,&rank);
    MPI_Comm_size(MPI_COMM_WORLD,&size);

    const char *input_file = NULL;

    for(int i=1;i<argc;i++){
        if(strcmp(argv[i],"-i")==0 && i+1<argc)
            input_file = argv[++i];
    }

    if(!input_file){
        if(rank==0)
            printf("Usage: mpirun -np P ./wariara_mpi -i file\n");
        MPI_Finalize();
        return 0;
    }

    if(rank==0)
        read_input(input_file);

    MPI_Bcast(&N,1,MPI_INT,0,MPI_COMM_WORLD);
    MPI_Bcast(E,MAXN*MAXN,MPI_INT,0,MPI_COMM_WORLD);

    double T_init = gettime() - t_init_start;

    int path[MAXN];

    for(int i=0;i<N;i++) visited[i]=false;
    visited[0]=true;
    path[0]=0;

    /* ───────── COMPUTATION ───────── */
    double t_comp_start = gettime();

    /* Distribute first-level branches cyclically */
    for(int city = 1 + rank; city < N; city += size)
    {
        for(int i=0;i<N;i++) visited[i]=false;
        visited[0]=true;
        visited[city]=true;
        path[0]=0;
        path[1]=city;

        branch_and_bound(2,
                         city,
                         E[0][city],
                         path);
    }

    /* ───────── GLOBAL REDUCTIONS ───────── */
    int global_best;
    MPI_Allreduce(&best_cost,
                  &global_best,
                  1,
                  MPI_INT,
                  MPI_MIN,
                  MPI_COMM_WORLD);

    long long global_completed;
    long long global_pruned;

    MPI_Reduce(&routes_completed,
               &global_completed,
               1,
               MPI_LONG_LONG,
               MPI_SUM,
               0,
               MPI_COMM_WORLD);

    MPI_Reduce(&routes_pruned,
               &global_pruned,
               1,
               MPI_LONG_LONG,
               MPI_SUM,
               0,
               MPI_COMM_WORLD);

    /* Determine which rank has the global best */
    struct { int cost; int rank; } local, global;
    local.cost = best_cost;
    local.rank = rank;
    MPI_Allreduce(&local, &global, 1, MPI_2INT, MPI_MINLOC, MPI_COMM_WORLD);

    /* Send best path from owner to rank 0 */
    if(rank == global.rank){
        MPI_Send(best_path, N, MPI_INT, 0, 0, MPI_COMM_WORLD);
    }
    if(rank == 0){
        MPI_Recv(best_path, N, MPI_INT, global.rank, 0, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
    }

    double T_comp  = gettime() - t_comp_start;
    double T_total = T_init + T_comp;

    /* ───────── OUTPUT ───────── */
    if(rank==0)
    {
        long long theoretical = factorial(N-1);
        long long routes_explored = global_completed + global_pruned;
        double pruning_pct = 100.0 * global_pruned / (double)routes_explored;

        printf("\n=== Wariara Freights Inc. – Optimal Route ===\n");
        printf("Cities          : %d\n", N);
        printf("Minimum energy  : %d kWh\n", global_best);
        printf("Optimal route   : ");
        for (int i=0; i<N; i++)
            printf("%d%s", best_path[i]+1, i<N-1?" -> ":"\n");

        printf("\n=== Branch-and-Bound Statistics ===\n");
        printf("Theoretical max : %lld (= (%d-1)! = %d!)\n",
               theoretical, N, N-1);
        printf("Routes completed: %lld\n", global_completed);
        printf("Subtrees pruned : %lld\n", global_pruned);
        printf("Total nodes hit : %lld\n", routes_explored);
        printf("Pruning rate    : %.2f%%\n", pruning_pct);

        printf("\n=== Timing ===\n");
        printf("Processes       : %d\n", size);
        printf("Initialisation  : %.6f s\n", T_init);
        printf("Computation     : %.6f s\n", T_comp);
        printf("Total           : %.6f s\n", T_total);
    }

    MPI_Finalize();
    return 0;
}
